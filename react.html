Exemplo: Uso de estado (state)


import React, { useState } from 'react';

function Contador() {
  const [count, setCount] = useState(0);
  return (
    <button onClick={() => setCount(count + 1)}>Cliquei {count} vezes</button>
  );
}

(Esse código precisa de um ambiente React para rodar.)
