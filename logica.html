Exemplo interativo: Testador de números (par, ímpar e fatorial)

Testador de Números

Código Exemplo em C


#include <stdio.h>

int fatorial(int n) {
  if(n <= 1) return 1;
  else return n * fatorial(n - 1);
}

int main() {
  int num;
  printf("Digite um número: ");
  scanf("%d", &num);
  if(num % 2 == 0) {
    printf("Número par\n");
  } else {
    printf("Número ímpar\n");
  }
  printf("Fatorial: %d\n", fatorial(num));
  return 0;
}

Código Exemplo em Python


def fatorial(n):
    if n <= 1:
        return 1
    else:
        return n * fatorial(n - 1)

num = int(input("Digite um número: "))
if num % 2 == 0:
    print("Número par")
else:
    print("Número ímpar")
print("Fatorial:", fatorial(num))
